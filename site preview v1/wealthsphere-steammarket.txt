// ROI Calc
function updateROI(){
  const sel=document.querySelector('#tab-roi select');
  const [sell,buy]=sel.value.split(',').map(Number);
  document.getElementById('roiBuy').value=buy;
  document.getElementById('roiSell').value=sell;
  calcROI();
}
 
function calcROI(){
  const buy=parseFloat(document.getElementById('roiBuy').value)||0;
  const sell=parseFloat(document.getElementById('roiSell').value)||0;
  const qty=parseInt(document.getElementById('roiQty').value)||1;
  const dateVal=document.getElementById('roiDate').value;
  const days=dateVal?Math.round((Date.now()-new Date(dateVal))/(86400000)):180;
  const commission=sell*0.15*qty;
  const lucro=(sell-buy)*qty;
  const liquido=lucro-commission;
  const roi=buy>0?((sell-buy)/buy*100):0;
  const anual=days>0?((Math.pow(1+roi/100,365/days)-1)*100):0;
  const fmt=v=>(v>=0?'+':'')+v.toFixed(2);
  const fmtE=v=>(v>=0?'+€':'-€')+Math.abs(v).toFixed(2);
  document.getElementById('r-lucro').textContent=fmtE(lucro);
  document.getElementById('r-lucro').className='roi-item-val '+(lucro>=0?'pos':'neg');
  document.getElementById('r-liquido').textContent=fmtE(liquido);
  document.getElementById('r-liquido').className='roi-item-val '+(liquido>=0?'pos':'neg');
  document.getElementById('r-roi').textContent=fmt(roi)+'%';
  document.getElementById('r-roi').className='roi-item-val '+(roi>=0?'pos':'neg');
  document.getElementById('r-anual').textContent=fmt(anual)+'%';
  document.getElementById('r-anual').className='roi-item-val '+(anual>=0?'pos':'neg');
  document.getElementById('r-dias').textContent=days+' dias';
  document.getElementById('r-comissao').textContent='€'+commission.toFixed(2);
}

}
 
function simAberturas(n){
  const ev=1.42,cost=3.06;
  const esperado=(ev*n).toFixed(2);
  const gasto=(cost*n).toFixed(2);
  const diff=(ev*n-cost*n).toFixed(2);
  const el=document.getElementById('simResult');
  el.style.display='block';
  el.innerHTML=`<strong>Simulação ${n} aberturas:</strong><br>Gasto: €${gasto} · Valor esperado: €${esperado} · <span style="color:var(--red)">Perda esperada: -€${Math.abs(diff)}</span>`;
}
 
// Charts
function drawChart(id,datasets,labels,height){
  const c=document.getElementById(id);if(!c)return;
  const ctx=c.getContext('2d');
  const W=c.offsetWidth,H=height||160;
  c.width=W*devicePixelRatio;c.height=H*devicePixelRatio;
  ctx.scale(devicePixelRatio,devicePixelRatio);
  const pad={t:10,r:10,b:22,l:42};
  const cW=W-pad.l-pad.r,cH=H-pad.t-pad.b;
  const allV=datasets.flatMap(d=>d.data);
  const mx=Math.max(...allV)*1.02,mn=Math.min(...allV)*0.98;
  const n=labels.length;
  const xOf=i=>pad.l+(i/(n-1))*cW;
  const yOf=v=>pad.t+cH-((v-mn)/(mx-mn))*cH;
  ctx.strokeStyle='rgba(90,74,58,0.07)';ctx.lineWidth=1;
  for(let g=0;g<=3;g++){
    const y=pad.t+(g/3)*cH;
    ctx.beginPath();ctx.moveTo(pad.l,y);ctx.lineTo(pad.l+cW,y);ctx.stroke();
    const v=mx-(g/3)*(mx-mn);
    ctx.fillStyle='rgba(138,122,106,0.65)';ctx.font='9px DM Sans';ctx.textAlign='right';
    ctx.fillText(v>=100?v.toFixed(0):'€'+v.toFixed(0),pad.l-4,y+3);
  }
  labels.forEach((l,i)=>{if(i%(Math.ceil(n/6))===0||i===n-1){ctx.fillStyle='rgba(138,122,106,0.65)';ctx.font='9px DM Sans';ctx.textAlign='center';ctx.fillText(l,xOf(i),H-4)}});
  datasets.forEach(({data,color})=>{
    const g=ctx.createLinearGradient(0,pad.t,0,pad.t+cH);
    g.addColorStop(0,color+'28');g.addColorStop(1,color+'04');
    ctx.beginPath();data.forEach((v,i)=>i?ctx.lineTo(xOf(i),yOf(v)):ctx.moveTo(xOf(i),yOf(v)));
    ctx.lineTo(xOf(n-1),pad.t+cH);ctx.lineTo(xOf(0),pad.t+cH);ctx.closePath();
    ctx.fillStyle=g;ctx.fill();
    ctx.beginPath();data.forEach((v,i)=>i?ctx.lineTo(xOf(i),yOf(v)):ctx.moveTo(xOf(i),yOf(v)));
    ctx.strokeStyle=color;ctx.lineWidth=2;ctx.lineJoin='round';ctx.stroke();
    ctx.beginPath();ctx.arc(xOf(n-1),yOf(data[n-1]),4,0,Math.PI*2);
    ctx.fillStyle=color;ctx.fill();ctx.strokeStyle='white';ctx.lineWidth=1.5;ctx.stroke();
  });
}
